  async calculateNextDueDate(currentDueDate: Date, rule: RecurrenceRule, originalTaskId: string): Promise<Date | null> {
    if (!currentDueDate || !rule) return null;
 
    // 1. Check 'after X occurrences' end condition first
    if (rule.endConditionType === 'afterOccurrences' && rule.count && rule.count > 0) {
      const occurrences = await this.getOccurrenceCount(originalTaskId);
      if (occurrences >= rule.count) {
        return null; // Max occurrences reached
      }
    }
 
    let nextDate = new Date(currentDueDate);
    const interval = rule.interval > 0 ? rule.interval : 1;
 
    switch (rule.frequency) {
      case 'daily':
        nextDate = addDays(nextDate, interval);
        break;
 
      case 'weekly':
        if (rule.daysOfWeek && rule.daysOfWeek.length > 0) {
          let candidateDate = new Date(nextDate);
          // Move to the start of the next interval block for searching
          candidateDate = addWeeks(candidateDate, interval -1); 
 
          // Search for the next valid day
          let attempts = 0; // Prevent infinite loop
          while (attempts < (7 * interval + 7)) { // Search within the interval window + one extra week
            candidateDate = addDays(candidateDate, 1);
            if (rule.daysOfWeek.includes(getDay(candidateDate))) {
              // Check if this date is truly after currentDueDate
              if (candidateDate > currentDueDate) {
                 nextDate = candidateDate;
                 break;
              }
            }
            attempts++;
          }
          if (attempts >= (7*interval + 7)) return null; // Should not happen if logic is correct
        } else {
          nextDate = addWeeks(nextDate, interval);
        }
        break;
 
      case 'monthly': {
        let monthCandidate = addMonths(new Date(currentDueDate), interval);
        if (rule.dayOfMonth) {
          if (rule.dayOfMonth > 0) {
            const potentialDate = setDate(monthCandidate, rule.dayOfMonth);
            // If setDate rolls over to next month (e.g. day 31 for Feb), use last day of target month
            if (getMonth(potentialDate) !== getMonth(monthCandidate)) {
              nextDate = endOfMonth(monthCandidate);
            } else {
              nextDate = potentialDate;
            }
          } else { // Negative dayOfMonth means from end of month
            const lastDay = endOfMonth(monthCandidate);
            nextDate = addDays(lastDay, rule.dayOfMonth + 1); // e.g., -1 means addDays(last, 0)
          }
        } else {
          nextDate = monthCandidate; // Day rolls over naturally if current day > days in target month
        }
        // Ensure the calculated date is after the current due date, especially for same-month calculations with dayOfMonth
        if (nextDate <= currentDueDate) {
            monthCandidate = addMonths(nextDate, interval); // Try one more interval if initial calculation is not in future
             if (rule.dayOfMonth) {
                if (rule.dayOfMonth > 0) {
                    const potentialDate = setDate(monthCandidate, rule.dayOfMonth);
                    if (getMonth(potentialDate) !== getMonth(monthCandidate)) nextDate = endOfMonth(monthCandidate); else nextDate = potentialDate;
                } else { nextDate = addDays(endOfMonth(monthCandidate), rule.dayOfMonth + 1); }
            } else { nextDate = monthCandidate; }
        }
        break;
      }
 
      case 'yearly': {
        let yearCandidate = addYears(new Date(currentDueDate), interval);
        if (rule.monthOfYear && rule.dayOfMonth) {
          const targetMonth = rule.monthOfYear - 1; // 0-indexed
          const potentialDate = setDate(setMonth(yearCandidate, targetMonth), rule.dayOfMonth);
          // If date rolled over (e.g. Feb 29 on non-leap year for monthOfYear=Feb, dayOfMonth=29)
          if (getMonth(potentialDate) !== targetMonth || getDate(potentialDate) !== rule.dayOfMonth) {
            nextDate = endOfMonth(setMonth(yearCandidate, targetMonth));
          } else {
            nextDate = potentialDate;
          }
        } else {
          nextDate = yearCandidate;
        }
         // Ensure the calculated date is after the current due date
        if (nextDate <= currentDueDate) {
            yearCandidate = addYears(nextDate, interval); // Try one more interval
            if (rule.monthOfYear && rule.dayOfMonth) {
                const targetMonth = rule.monthOfYear - 1;
                const potentialDate = setDate(setMonth(yearCandidate, targetMonth), rule.dayOfMonth);
                if (getMonth(potentialDate) !== targetMonth || getDate(potentialDate) !== rule.dayOfMonth) nextDate = endOfMonth(setMonth(yearCandidate, targetMonth)); else nextDate = potentialDate;
            } else { nextDate = yearCandidate; }
        }
        break;
      }
      default:
        return null; // Unknown frequency
    }
 
    // 2. Check 'onDate' end condition
    if (rule.endConditionType === 'onDate' && rule.endDate) {
      if (nextDate > rule.endDate) {
        return null; // Next date is past the end date
      }
    }
 
    // Ensure the final nextDate is actually after the currentDueDate
    if (nextDate <= currentDueDate) {
      // This can happen if interval logic is complex or currentDueDate is very close to a boundary
      // For safety, try to advance by one minimal step of the frequency if it's still not in the future
      // This is a fallback and ideally the main logic should prevent this.
      console.warn('Calculated nextDueDate is not after currentDueDate. Attempting fallback.', {currentDueDate, nextDate, rule});
      switch (rule.frequency) {
        case 'daily': nextDate = addDays(nextDate, 1); break;
        case 'weekly': nextDate = addWeeks(nextDate, 1); break;
        case 'monthly': nextDate = addMonths(nextDate, 1); break;
        case 'yearly': nextDate = addYears(nextDate, 1); break;
      }
      // Re-check end date after fallback adjustment
      if (rule.endConditionType === 'onDate' && rule.endDate && nextDate > rule.endDate) return null;
    }
    
    // Final check: if somehow nextDate is still not in the future, something is wrong.
    if (nextDate <= currentDueDate) return null;
 
    return nextDate;
